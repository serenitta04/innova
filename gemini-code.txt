        /* --- 1. ENRUTADOR LOGICO INTERNO (SPA ROUTING EFFECT) --- */

        /* --- 2. MOTOR GRÁFICO DE NODOS (CANVAS NETWORK) --- */
        const canvas = document.getElementById('hero-canvas');
        const ctx = canvas.getContext('2d');
        let totalParticles = [];
        const maxParticlesCount = 60;

        function adaptCanvasSize() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', adaptCanvasSize);
        adaptCanvasSize();

        class NetworkParticle {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.vx = (Math.random() - 0.5) * 0.3;
                this.vy = (Math.random() - 0.5) * 0.3;
                this.size = Math.random() * 1.5 + 1;
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                if (this.x < 0 || this.x > canvas.width) this.vx *= -1;
                if (this.y < 0 || this.y > canvas.height) this.vy *= -1;
            }
            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fillStyle = 'rgba(56, 189, 248, 0.25)';
                ctx.fill();
            }
        }

        for (let i = 0; i < maxParticlesCount; i++) {
            totalParticles.push(new NetworkParticle());
        }

        function executionLoop() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            for (let i = 0; i < totalParticles.length; i++) {
                totalParticles[i].update();
                totalParticles[i].draw();
                
                // Cálculo de enlaces de proximidad de redes
                for (let j = i + 1; j < totalParticles.length; j++) {
                    let dx = totalParticles[i].x - totalParticles[j].x;
                    let dy = totalParticles[i].y - totalParticles[j].y;
                    let distance = Math.sqrt(dx * dx + dy * dy);
                    if (distance < 130) {
                        ctx.beginPath();
                        ctx.moveTo(totalParticles[i].x, totalParticles[i].y);
                        ctx.lineTo(totalParticles[j].x, totalParticles[j].y);
                        ctx.strokeStyle = `rgba(37, 99, 235, ${0.12 - (distance / 130) * 0.12})`;
                        ctx.lineWidth = 0.5;
                        ctx.stroke();
                    }
                }
            }
            requestAnimationFrame(executionLoop);
        }
        executionLoop();

        /* --- 3. PROCESAMIENTO DE ESTADOS: ESCAPE ROOM ACADÉMICO --- */
